31class SoundcloudDownloader: 

32 """ 

33 Downloads tracks from SoundCloud accounts listed in the database. 

34 

35 Features: 

36 - Downloads only MP3 files that are not yet in the archive. 

37 - Embeds metadata and optionally thumbnails using FFmpeg. 

38 - Supports private/follower-only tracks using session cookies. 

39 - Handles batch downloads with configurable parallelism and throttling. 

40 - Skips tracks outside a configurable duration range. 

41 """

85 def _match_filter(self, info): 

86 duration = info.get("duration") 

87 title = info.get("title", "unknown") 

88 if not duration or duration < 60 or duration > 21600: 

89 logging.info(f"Skipping track '{title}' (duration: {duration}s)") 

90 return "Outside allowed duration range"

93 def download_account(self, name: str, yt_dl_opts: dict = None): 

94 link = f"http://soundcloud.com/{name}/tracks" 

95 logging.info(f"Downloading from SoundCloud account: {name}") 

96 

97 for attempt in range(1, 4): 

98 try: 

99 with YoutubeDL(yt_dl_opts) as ydl: 

100 ydl.add_post_processor(FFmpegMetadataPP(ydl)) 

101 ydl.add_post_processor(EmbedThumbnailPP(ydl)) 

102 ydl.add_post_processor(SoundcloudSongProcessor()) 

103 ydl.download([link]) 

104 logging.info(f"Finished downloading from: {name}") 

105 return 

106 except Exception as e: 

107 msg = str(e) 

108 if '403' in msg: 

109 wait_time = random.randint(60, 300) 

110 logging.warning(f"403 Forbidden for {name}. Pausing {wait_time}s before retry...") 

111 time.sleep(wait_time) 

112 elif '404' in msg: 

113 logging.info(f"Got 404 for {name} — skip song.") 

114 elif 'already in the archive' in msg: 

115 logging.info(f"All tracks for {name} already in archive — skipping.") 

116 return # ✅ don't retry 

117 else: 

118 logging.warning(f"Attempt {attempt} failed for {name}: {e}", exc_info=True) 

119 time.sleep(5 * attempt) 

120 

121 logging.error(f"SoundCloud download failed for {name} after 3 attempts.")

139 batch_num = i // self.burst_size + 1 

140 logging.info(f"Processing batch {batch_num} of {total_batches}")

143 futures = {} 

144 for acc in batch: 

145 # Clone ydl_opts and set archive file (shared or per-account)

154 futures[executor.submit(self.download_account, acc, ydl_opts)] = acc

156 for future in concurrent.futures.as_completed(futures): 

157 acc = futures[future] 

158 processed += 1 

159 job_manager.publish({

162 "current": processed,

165
