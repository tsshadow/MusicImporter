36class BaseSong: 

37 """ 

38 Represents a single audio file and its associated metadata. 

39 

40 Supports MP3, FLAC, WAV, M4A, and AAC (partial). 

41 Provides methods to read, clean, update, and save tags. 

42 """

44 def __init__(self, path, extra_info=None): 

45 """ 

46 Initializes the BaseSong object by loading the file and its tags. 

47 

48 Args: 

49 path (str): Path to the audio file. 

50 """ 

51 self.rules: list[TagRule] = [] 

52 paths = path.rsplit(s.delimiter, 2) 

53 self._path: str = path 

54 self._filename = str(paths[-1]) 

55 self._extension = os.path.splitext(self._filename)[1].lower() 

56 

57 music_file_classes = { 

58 ".mp3": lambda p: (MP3(p, ID3=EasyID3), MusicFileType.MP3), 

59 ".flac": lambda p: (FLAC(p), MusicFileType.FLAC), 

60 ".wav": lambda p: (WAVE(p), MusicFileType.WAV), 

61 ".m4a": lambda p: (MP4(p), MusicFileType.M4A) 

62 # ".aac": lambda p: (APEv2(p), MusicFileType.AAC) # Non-standard fallback 

63 } 

64 

65 try: 

66 self.music_file, self.type = music_file_classes[self._extension](path) 

67 except KeyError: 

68 raise ExtensionNotSupportedException(f"{self._extension} is not supported") 

69 except MP4StreamInfoError: 

70 print("MP4StreamInfoError" + path) 

71 try: 

72 os.remove(path) 

73 print("deleted broken file: " + path) 

74 log_broken_file(path) 

75 return 

76 except Exception as e: 

77 print("Kon bestand niet verwijderen:", e) 

78 

79 if self.type == MusicFileType.WAV and self.music_file.tags is None: 

80 self.music_file.add_tags() 

81 

82 self.tag_collection = TagCollection( 

83 self.music_file.tags if self.type != MusicFileType.AAC else self.music_file 

84 ) 

85 

86 if extra_info: 

87 self._apply_extra_info(extra_info) 

88 

89 if self.type == MusicFileType.FLAC: 

90 NormalizeFlacTagsRule().apply(self) 

91 if analyze_bpm: 

92 self.rules.append(AnalyzeBpmRule())

101 if self.music_file and self.music_file.get(tag):

120 def save_file(self): 

121 if not hasattr(self, 'tag_collection'): 

122 return 

123 

124 changes = False 

125 for tag in self.tag_collection.get().values(): 

126 if isinstance(tag, Tag) and tag.has_changes(): 

127 self.set_tag(tag) 

128 changes = True 

129 if changes: 

130 logging.info(f"File saved: {self.path()}") 

131 self.music_file.save()

133 def set_tag(self, tag: Tag): 

134 """Sets a tag on the underlying music file based on type.""" 

135 logging.info(f"Set tag {tag.tag} to {tag.to_string()} (was: {self.music_file.get(tag.tag)})") 

136 

137 value = tag.to_string() 

138 

139 if self.type == MusicFileType.MP3: 

140 self.music_file[tag.tag] = value 

141 

142 elif self.type == MusicFileType.FLAC: 

143 flac_key = FLACTags.get(tag.tag) 

144 if flac_key: 

145 self.music_file[flac_key] = value 

146 

147 elif self.type == MusicFileType.WAV: 

148 wav_key = WAVTags.get(tag.tag) 

149 if wav_key: 

150 self.music_file.tags[wav_key] = mutagen.id3.TextFrame(encoding=3, text=[value]) 

151 

152 elif self.type == MusicFileType.M4A: 

153 mp4_key = MP4Tags.get(tag.tag) 

154 if mp4_key: 

155 self.music_file.tags[mp4_key] = value 

156 else: 

157 # Fallback for LMS-style custom tags 

158 from mutagen.mp4 import MP4FreeForm 

159 custom_key = f"----:com.apple.iTunes:{tag.tag.upper()}" 

160 self.music_file.tags[custom_key] = [MP4FreeForm(value.encode('utf-8'))]

172 """

175 # Prefer "artists", fallback to "artist", then "uploader" 

176 artist = (
