10class InferRemixerFromTitleRule(TagRule): 

11 """ 

12 Haalt remixer/edit-artiesten uit de titel (bijv. '(XYZ Remix)') en voegt ze toe 

13 aan het ARTIST-veld én het aparte REMIXERS-tagveld (nieuw).

17 BRACKET_RE = re.compile(r"\(([^()]*)\)") 

18 SUFFIX_CLEANUP_FULL = re.compile( 

19 r"\s*\b(" 

20 r"album|bootleg|cinematic|climax|cut|dub|dubstep|edit|extended|hardcore|hardstyle|instrumental|kick|live|mix|non vocal|non-vocal|nostalgia|old school|original|pro remix|radio|refix|remastered|remix|re-kick|uptempo|version|vip|vocal|" 

21 r"\d{4}|2k\d{2}" 

22 r")\b\s*$", 

23 re.IGNORECASE 

24 ) 

25 SUFFIX_CLEANUP_SIMPLE = re.compile(r"\s*\b(edit|remix|refix|bootleg)\b\s*$", re.IGNORECASE)

27 def __init__(self, artist_db=None, ignored_db=None, ask_for_missing: bool = False):

32 def _clean_artist_name(self, name: str) -> str: 

33 name = re.sub(r"['’]s\s+(remix|edit|refix|version|bootleg|mix|vip)\b", r" \1", name, flags=re.IGNORECASE) 

34 while True: 

35 new_name = re.sub(self.SUFFIX_CLEANUP_FULL, "", name).strip() 

36 if new_name == name: 

37 return new_name 

38 name = new_name 

39 

40 def _clean_suffix_only(self, name: str) -> str: 

41 while True: 

42 new_name = re.sub(self.SUFFIX_CLEANUP_SIMPLE, "", name).strip() 

43 if new_name == name: 

44 return new_name 

45 name = new_name
